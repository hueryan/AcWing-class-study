```sh
cd app/
mkdir match_system

cd match_system
touch __init__/py
mkdir thrift  # 存储thrift配置文件
mkdir src  # 具体逻辑
vim thrift/match.thrift

cd src 
thrift --gen py ../thrift/match.thrift
mv gen-py/ match_server
touch __init__.py
vim main.py  # MatchServer
chmod +x main.py

cd ~/app/game/consumers/multiplayer  
vim index.py  # 客户端    当服务端接收到create_player请求时，相匹配池发送申请

vim ~app/app/asgi.py  # 爆一个API


vim ~/app/game/models/player/player.py  # 扩容数据库
python3 ~/app/manage.py makemigrations  # 更新数据库
python3 ~/app/manage.py migrate

vim cd ~/app/game/consumers/multiplayer/index.py  # 从数据库导入score
```
